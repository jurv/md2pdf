  \AddToShipoutPictureBG*{%
    \AtPageLowerLeft{%
      \parbox[b][\paperheight][c]{\paperwidth}{\centering #1}%
    }%
  }%

\newcommand{\mdtwoaddcoverimagecover}[1]{%
  \sbox{\mdtwocoverimgbox}{\includegraphics{#1}}%
  \edef\mdtwo@img@ratio{\fpeval{\strip@pt\wd\mdtwocoverimgbox / \strip@pt\dimexpr\ht\mdtwocoverimgbox+\dp\mdtwocoverimgbox\relax}}%
  \edef\mdtwo@page@ratio{\fpeval{\strip@pt\paperwidth / \strip@pt\paperheight}}%
  \ifdim\mdtwo@img@ratio pt>\mdtwo@page@ratio pt
    \mdtwoplacecovergraphic{\begingroup\setkeys{Gin}{width=10000pt,height=\paperheight,keepaspectratio=true}\includegraphics{#1}\endgroup}%
  \else
    \mdtwoplacecovergraphic{\begingroup\setkeys{Gin}{width=\paperwidth,height=10000pt,keepaspectratio=true}\includegraphics{#1}\endgroup}%
  \fi
}
\makeatother
